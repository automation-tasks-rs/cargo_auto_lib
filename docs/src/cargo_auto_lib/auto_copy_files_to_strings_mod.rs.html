5// trait must be in scope
6use base64ct::Encoding;
7
8use crate::error_mod::{Error, Result};
9use crate::public_api_mod::{RESET, YELLOW};

11// region: auto_md_to_doc_comments include doc_comments/copy_folder_files_into_module.md A ///
12/// Copy all files from the folder into a module as strings (static &str).
13///
14/// The Rust code to modify has the markers:
15///
16/// ```Rust ignore
17/// //comment region: files copied into strings by automation tasks
18///
19/// //comment endregion: files copied into strings by automation tasks
20///
21/// ```
22///
23/// In this instructions I changed `[//]` to `[//comment]` to not process these markers.
24///
25/// First we create the complete text, then we check if the old text needs to be replaced.
26///
27/// Binary files need a special treatment:
28///
29/// ```Rust ignore
30/// ext_for_binary_files=vec![".ico",".jpg",".png",".woff2"];
31/// ```
32///
33/// Exclude big folders:
34///
35/// ```Rust ignore
36/// exclude_big_folders = vec!["/.git","/target","/docs"];
37/// ```
38///
39// endregion: auto_md_to_doc_comments include doc_comments/copy_folder_files_into_module.md A ///
40pub fn copy_folder_files_into_module(
41    folder_path: &std::path::Path,
42    module_path: &std::path::Path,
43    ext_for_binary_files: &[&str],
44    exclude_big_folders: &[String],
45) -> Result<()> {
46    let folder_path = camino::Utf8Path::from_path(folder_path).ok_or_else(|| Error::ErrorFromStr("folder_path is None"))?;
47    let module_path = camino::Utf8Path::from_path(module_path).ok_or_else(|| Error::ErrorFromStr("module_path is None"))?;
48
49    println!("  {YELLOW}copy_folder_files_into_module {folder_path}, {module_path}{RESET}");
50    // traverse and get all file_names
51    let files = crate::traverse_dir_with_exclude_dir(folder_path.as_std_path(), "", exclude_big_folders)?;
52    let mut new_code = String::new();
53    for file_name in files.iter() {
54        let file_name_short = file_name.trim_start_matches(&format!("{folder_path}/"));
55        // avoid Cargo.lock file
56        if file_name_short == "Cargo.lock" {
57            continue;
58        }
59        // let the user define in an input parameter what files are binaries and not text.
60        let mut is_binary_file = false;
61        for x in ext_for_binary_files.iter() {
62            if file_name_short.ends_with(x) {
63                is_binary_file = true;
64                break;
65            }
66        }
67
68        let file_content = if is_binary_file {
69            // convert binary file to base64
70            let b = std::fs::read(file_name)?;
71            base64ct::Base64::encode_string(&b)
72        } else {
73            // all others are text files
74            std::fs::read_to_string(file_name)?
75        };
76
77        new_code.push_str(&format!(
78            r####"vec_file.push(crate::FileItem{{
79            file_name :"{}",
80            file_content : r###"{}"###,
81}});    
82"####,
83            &file_name_short, &file_content
84        ));
85    }
86
87    // read the content of the module, delimited by markers
88    let module_content = std::fs::read_to_string(module_path)?;
89    let start_pos =
90        crate::find_pos_start_data_after_delimiter(&module_content, 0, "// region: files copied into strings by automation tasks\n")
91            .expect("didn't find // region: files copied..");
92    let end_pos =
93        crate::find_pos_end_data_before_delimiter(&module_content, 0, "// endregion: files copied into strings by automation tasks")
94            .expect("didn't find // endregion: files copied..");
95    let old_code = &module_content[start_pos..end_pos];
96
97    // compare the text, if different replace
98    if old_code != new_code {
99        let mut new_module_content = String::new();
100        new_module_content.push_str(&module_content[..start_pos]);
101        new_module_content.push_str(&new_code);
102        new_module_content.push_str(&module_content[end_pos..]);
103        std::fs::write(module_path, &new_module_content)?;
104    }
105    Ok(())
106}
