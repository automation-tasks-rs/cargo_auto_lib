14/// Struct that contains 4 types of lines count: code, doc comments, comments, test and examples.
15pub struct LinesOfCode {
16    /// lines with code in srs files
17    pub src_code_lines: usize,
18    /// lines with doc_comments in srs files
19    pub src_doc_comment_lines: usize,
20    /// lines with comments in srs files
21    pub src_comment_lines: usize,
22    /// unit plus integration tests
23    pub tests_lines: usize,
24    /// all lines in examples files
25    pub examples_lines: usize,
26}

28// region: auto_md_to_doc_comments include doc_comments/auto_lines_of_code.md A ///
29/// This function inserts shield badges with lines_of_code into README.rs.  
30///
31/// ![auto_lines_of_code.png](https://github.com/automation-tasks-rs/cargo_auto_lib/blob/main/images/auto_lines_of_code.png?raw=true)
32///
33/// The parameter Link will be used for shield badge. If empty_string, the git remote repository will be used.  
34/// Lines of code are not a "perfect" measurement of anything.  
35/// Anybody can write a very big number of lines of useless code and comments.  
36/// But for 95% of the cases they are good enough.  
37/// Most of the developers use some "standard" coding practices and that is quantifiable and comparable.  
38///
39/// The `src_code_lines` is the most important count.  
40/// That is actual code written for that project without doc comments, comments, unit tests, integration tests and examples.  
41/// Sometimes is great to see a big number here. It means there was a lot of work invested. But other times we want to see a small number. It
42/// means the developer understands the problem very well and don't try to solve anything outside that scope.  
43/// The `src_doc_comment_lines` counts doc comments. They will eventually become docs. The count of lines shows how many documentation is written.  
44/// The `src_comment_lines` counts code comments. Code comments are important to understand the code. The count of lines shows how understandable is the code.  
45/// The `tests_lines` counts lines in tests and shows how good is the code tested. Here are the unit tests and integration test combined.  
46/// The `examples_lines` counts lines in examples and shows how good is explained how to use the code.  
47///
48///
49/// ## Folder and file structure
50///
51/// The folder structure of a single Rust project is simple.  
52/// The project starts in the folder that contains `Cargo.toml`.  
53/// The `src/` folder contains all the rust `*.rs` files.  
54/// The `tests/` folder contains integration tests.  
55/// The `examples/` folder contains examples.  
56/// Some rs files can be excluded from the count adding this line near the start of the file: // exclude from auto_lines_of_code
57/// Inside a rs file the doc comment line start with `///` or `//!`.  
58/// The normal comments start with `//` or `/!`.  
59/// I will ignore the block comments. They are usually NOT used for comments, but to temporarily disable a piece of code. So I count this as code and not comments.  
60/// The `src/*.rs` file can contain unit tests that start with `#[cfg(test)]`. I assume that these are always at the end of the file.  
61/// There should not be any normal code after `#[cfg(test)]`, only tests.  
62/// All other files: `md`, `toml`, `html`, `js`, ... are not counted.  
63///
64/// ## Include into README.md
65///
66/// If the README.md file contains these markers (don't copy the numbers 1 and 2):  
67///
68/// ```md
69/// [//comment]: # (auto_lines_of_code start)
70///
71/// [//comment]: # (auto_lines_of_code end)
72/// ```
73///
74/// In this instructions I changed `[//]` to `[//comment]` to not process these markers.
75///
76/// The function will include the shield badges code between them.  
77/// It will erase the previous content.  
78/// Use git diff to see the change.  
79///
80// endregion: auto_md_to_doc_comments include doc_comments/auto_lines_of_code.md A ///
81pub fn auto_lines_of_code(link: &str) -> Result<()> {
82    println!("  {YELLOW}Running auto_lines_of_code{RESET}");
83    let link = if link.is_empty() {
84        crate::auto_git_mod::process_git_remote()
85    } else {
86        link.to_string()
87    };
88    // Cargo.toml contains the list of projects
89    let lines_of_code = count_lines_of_code()?;
90    let text_to_include = to_string_as_shield_badges(&lines_of_code, &link);
91    include_into_readme_md(&text_to_include)?;
92    println!("  {YELLOW}Finished auto_lines_of_code{RESET}");
93    Ok(())
94}

96/// Returns the counted lines of code
97///
98/// Does not write to README.md.
99pub fn count_lines_of_code() -> Result<LinesOfCode> {
100    let mut lines_of_code = LinesOfCode::default();
101
102    // src folder
103    let files = crate::utils_mod::traverse_dir_with_exclude_dir(
104        camino::Utf8Path::new("src").as_std_path(),
105        "/*.rs",
106        // avoid big folders
107        &["/.git".to_string(), "/target".to_string(), "/docs".to_string()],
108    )?;
109    // println!("{:#?}", files);
110    for rs_file_name in files.iter() {
111        // Open the file in read-only mode (ignoring errors).
112        let file = File::open(rs_file_name)?;
113        let reader = BufReader::new(file);
114        let mut is_unit_test = false;
115        // Read the file line by line using the lines() iterator from std::io::BufRead.
116        for line in reader.lines() {
117            let line = line?; // Ignore errors.
118            let line = line.trim_start();
119            if line == "// exclude from auto_lines_of_code" {
120                break;
121            }
122            if line.starts_with("///") || line.starts_with("//!") {
123                lines_of_code.src_doc_comment_lines += 1;
124            } else if line.starts_with("//") || line.starts_with("/!") {
125                lines_of_code.src_comment_lines += 1;
126            } else if line.starts_with("#[cfg(test)]") {
127                is_unit_test = true;
128            } else if is_unit_test {
129                lines_of_code.tests_lines += 1;
130            } else {
131                lines_of_code.src_code_lines += 1;
132            }
133        }
134    }
135    // tests folder
136    let files = crate::utils_mod::traverse_dir_with_exclude_dir(
137        camino::Utf8Path::new("tests").as_std_path(),
138        "/*.rs",
139        // avoid big folders
140        &["/.git".to_string(), "/target".to_string(), "/docs".to_string()],
141    )?;
142    // println!("{:#?}", files);
143    for rs_file_name in files.iter() {
144        // Open the file in read-only mode (ignoring errors).
145        let file = File::open(rs_file_name)?;
146        let reader = BufReader::new(file);
147        // Read the file line by line using the lines() iterator from std::io::BufRead.
148        for _line in reader.lines() {
149            lines_of_code.tests_lines += 1;
150        }
151    }
152
153    // examples folder
154    let files = crate::utils_mod::traverse_dir_with_exclude_dir(
155        camino::Utf8Path::new("examples").as_std_path(),
156        "/*.rs",
157        // avoid big folders
158        &["/.git".to_string(), "/target".to_string(), "/docs".to_string()],
159    )?;
160    for rs_file_name in files.iter() {
161        // Open the file in read-only mode (ignoring errors).
162        let file = File::open(rs_file_name)?;
163        let reader = BufReader::new(file);
164        // Read the file line by line using the lines() iterator from std::io::BufRead.
165        for _line in reader.lines().enumerate() {
166            lines_of_code.examples_lines += 1;
167        }
168    }
169    //println!("{:#?}", &lines_of_code);
170    // return
171    Ok(lines_of_code)
172}

174/// Returns a string with the markdown code for 4 shield badges.
175///
176/// Every badge has the link to the url given as first parameter
177/// or automatically finds out the GitHub git remote repository url.
178///
179/// let v = cargo_auto_lib::auto_lines_of_code_mod::count_lines_of_code();
180/// let badges = cargo_auto_lib::auto_lines_of_code_mod::to_string_as_shield_badges(&v,"");
181///
182/// println!("{}", badges);
183/// TODO: cargo-auto_lib could change the code to make some element visibility `pub` only for testing. And after return to normal.  
184fn to_string_as_shield_badges(v: &LinesOfCode, link: &str) -> String {
185    //println!("to_string_as_shield_badges() start");
186
187    let src_code_lines = format!(
188        "[![Lines in Rust code](https://img.shields.io/badge/Lines_in_Rust-{}-green.svg)]({})",
189        v.src_code_lines, link
190    );
191    let src_doc_comment_lines = format!(
192        "[![Lines in Doc comments](https://img.shields.io/badge/Lines_in_Doc_comments-{}-blue.svg)]({})",
193        v.src_doc_comment_lines, link
194    );
195    let src_comment_lines = format!(
196        "[![Lines in Comments](https://img.shields.io/badge/Lines_in_comments-{}-purple.svg)]({})",
197        v.src_comment_lines, link
198    );
199    let example_lines = format!(
200        "[![Lines in examples](https://img.shields.io/badge/Lines_in_examples-{}-yellow.svg)]({})",
201        v.examples_lines, link
202    );
203    let tests_lines = format!(
204        "[![Lines in tests](https://img.shields.io/badge/Lines_in_tests-{}-orange.svg)]({})",
205        v.tests_lines, link
206    );
207    //return
208    format!(
209        "{}\n{}\n{}\n{}\n{}\n",
210        src_code_lines, src_doc_comment_lines, src_comment_lines, example_lines, tests_lines
211    )
212}

214/// Includes (writes, modifies) the shield badge code into README.md file
215fn include_into_readme_md(include_str: &str) -> Result<()> {
216    let start_delimiter = "[//]: # (auto_lines_of_code start)";
217    let end_delimiter = "[//]: # (auto_lines_of_code end)";
218    let file_name = "README.md";
219
220    if let Ok(readme_content) = std::fs::read_to_string(file_name) {
221        // check if file have CRLF instead of LF and show error
222        if readme_content.contains("\r\n") {
223            return Err(Error::ErrorFromString(format!(
224                "{RED}Error: {} has CRLF line endings instead of LF. Correct the file! {RESET}",
225                file_name
226            )));
227        }
228
229        let mut new_readme_content = String::with_capacity(readme_content.len());
230        if let Some(mut pos_start) = readme_content.find(start_delimiter) {
231            pos_start += start_delimiter.len();
232            if let Some(pos_end) = readme_content.find(end_delimiter) {
233                new_readme_content.push_str(&readme_content[..pos_start]);
234                new_readme_content.push('\n');
235                new_readme_content.push_str(include_str);
236                new_readme_content.push('\n');
237                new_readme_content.push_str(&readme_content[pos_end..]);
238                /*
239                println!(
240                    "{}include_into_readme_md write file: {}{}",
241                    *GREEN, file_name, *RESET
242                );
243                 */
244                std::fs::write(file_name, new_readme_content)?;
245            }
246        }
247    }
248    Ok(())
