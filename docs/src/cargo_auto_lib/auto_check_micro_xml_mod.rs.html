3//! Checks the correctness of micro XML files.
4
5use crate::error_mod::{Error, Result};
6use crate::public_api_mod::{RED, RESET, YELLOW};
7use glob::glob;
8use reader_for_microxml::{ReaderForMicroXml, Token};
9
10/// I want html pages to be correct microXML when I use them for single page application.
11/// Before build or release this function will check for correctness.
12pub fn auto_check_micro_xml(path_to_html_pages: &str) -> Result<()> {
13    println!("  {YELLOW}Running auto_check_micro_xml {path_to_html_pages}{RESET}");
14    let glob_str = format!("{}/*.html", path_to_html_pages.trim_end_matches('/'));
15    // find html pages for single page application
16    for filename_result in glob(&glob_str)? {
17        let filename_pathbuff = filename_result?;
18        let filename_pathbuff = camino::Utf8Path::from_path(&filename_pathbuff).ok_or_else(|| Error::ErrorFromStr("filename is None"))?;
19
20        let file_name = filename_pathbuff
21            .file_name()
22            .ok_or_else(|| Error::ErrorFromStr("filename is None"))?;
23        let str_xml = std::fs::read_to_string(filename_pathbuff)?;
24
25        // check if file have CRLF instead of LF and show error
26        if str_xml.contains("\r\n") {
27            return Err(Error::ErrorFromString(format!(
28                "{RED}Error: {filename_pathbuff} has CRLF line endings instead of LF. Correct the file! {RESET}"
29            )));
30        }
31
32        // check microxml correctness. Error on errors.
33        check_micro_xml(&str_xml, file_name)?;
34    }
35    println!("  {YELLOW}Finished auto_check_micro_xml{RESET}");
36    Ok(())
37}

39/// Errors if the microXML string is not correct
40fn check_micro_xml(str_xml: &str, file_name: &str) -> Result<()> {
41    println!("  {YELLOW}Check MicroXml: {file_name}{RESET}");
42    // remove <!DOCTYPE html> because it is not microXML
43    let str_xml = str_xml.replace("<!DOCTYPE html>", "");
44    let reader_iterator = ReaderForMicroXml::new(&str_xml);
45    // reader_iterator is iterator Option<Result<Token,&str>>
46    // the first option is used for the iterator to know where is the end
47    // then the Result can have an Token or an Error
48    let mut vec_elem: Vec<&str> = vec![];
49    for result_token in reader_iterator {
50        match result_token {
51            Ok(token) => match token {
52                Token::StartElement(name) => vec_elem.push(name),
53                Token::Attribute(_name, _value) => continue,
54                Token::TextNode(_txt) => continue,
55                Token::Comment(_txt) => continue,
56                Token::EndElement(end_element_name) => {
57                    let start_element_name = vec_elem.pop().ok_or_else(|| Error::ErrorFromStr("vec_elem.pop() is None"))?;
58                    if !end_element_name.is_empty() && end_element_name != start_element_name {
59                        return Err(Error::ErrorFromString(format!(
60                            "{RED}MicroXml {} start {} does not match end {}{RESET}",
61                            file_name, start_element_name, end_element_name,
62                        )));
63                    }
64                }
65            },
66            Err(err_msg) => {
67                return Err(Error::ErrorFromString(format!(
68                    "{RED}MicroXml {} incorrect : {}{RESET}",
69                    file_name, err_msg,
70                )));
71            }
72        }
73    }
74    Ok(())
75}
