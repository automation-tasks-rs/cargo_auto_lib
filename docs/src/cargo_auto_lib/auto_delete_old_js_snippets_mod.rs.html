12/// Deletes old js snippets when working with wasm-pack.  
13///
14/// The old folders for `js snippets` are not automatically deleted on building with `wasm-pack`.  
15/// This utils do that.  
16/// The function must be executed in the root project folder where is the Cargo.toml.  
17pub fn auto_delete_old_js_snippets() -> Result<()> {
18    let current_dir = std::env::current_dir()?;
19    let snippets_dir = current_dir.join("pkg").join("snippets");
20    //the first folder can be None
21    let mut opt_first_folder: Option<std::path::PathBuf> = None;
22    let mut opt_first_mtime: Option<FileTime> = None;
23
24    //find the newer folder and remove the older folder
25    //but not with dodrio_xxx name
26    for entry in std::fs::read_dir(snippets_dir)? {
27        let entry = entry?;
28        let second_folder = entry.path();
29        let second_name = entry
30            .file_name()
31            .into_string()
32            .map_err(|_| Error::ErrorFromStr("file_name into_string error"))?
33            .to_lowercase();
34        if !second_name.starts_with("dodrio") {
35            //println!("{:?}",second_folder);
36            let second_metadata = std::fs::metadata(&second_folder)?;
37            let second_mtime = FileTime::from_last_modification_time(&second_metadata);
38            //println!("{:?}",second_mtime);
39
40            match opt_first_mtime {
41                None => {
42                    opt_first_folder = Some(second_folder.clone());
43                    opt_first_mtime = Some(second_mtime);
44                }
45                Some(first_mtime) => match second_mtime.cmp(&first_mtime) {
46                    // if second_mtime > first_mtime {
47                    std::cmp::Ordering::Greater => {
48                        let first_folder = opt_first_folder.ok_or_else(|| Error::ErrorFromStr("opt_first_folder is None"))?;
49                        println!("  {YELLOW}delete first: {:?}{RESET}", first_folder);
50                        std::fs::remove_dir_all(first_folder)?;
51
52                        opt_first_folder = Some(second_folder.clone());
53                        opt_first_mtime = Some(second_mtime);
54                    }
55                    //  } else if first_mtime > second_mtime {
56                    std::cmp::Ordering::Less => {
57                        println!("  {YELLOW}delete second: {:?}{RESET}", second_folder);
58                        std::fs::remove_dir_all(second_folder)?;
59                    }
60                    // else
61                    std::cmp::Ordering::Equal => {
62                        eprintln!("{RED}Error: folders have the same date?{RESET}");
63                    }
64                },
65            }
66        }
67    }
68    Ok(())
69}
