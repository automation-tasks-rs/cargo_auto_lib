11/// Pretty HTML for docs.  
12///
13/// The HTML generated by `cargo doc` is ugly and difficult to `git diff`.
14/// Tidy HTML is a HTML checker and formatter installed on most Linuxes.
15pub fn auto_doc_tidy_html() -> Result<()> {
16    // First we check if tidy is installed on the system
17    // Run a dummy command and write the std/err output to tidy_warnings.txt.
18    // The command `2>` will overwrite the file and not append like `2>>`.
19    crate::run_shell_command_static("tidy xxx 2> docs/tidy_warnings.txt")?;
20    // Check if it contains `command not found`
21    let text = std::fs::read_to_string("docs/tidy_warnings.txt")?;
22    // don't need this file anymore
23    crate::run_shell_command_static("rm -f docs/tidy_warnings.txt")?;
24    if !text.contains("command not found") {
25        // Use tidy HTML to format the docs/*.html files to be human readable and usable for git diff.
26        // Options: -m modify file, -q quiet suppress nonessential output, -w wrap at 160, -i indent 2 spaces
27        // The warnings and errors are appended to the file docs/tidy_warnings.txt
28        crate::run_shell_command_static(
29            r#"find ./docs -name '*.html' -type f -print -exec tidy -mq --tidy-mark no -w 160 -i 2 '{}' \; >> docs/tidy_warnings.txt 2>&1 "#,
30        )?;
31    }
32    Ok(())
33}
