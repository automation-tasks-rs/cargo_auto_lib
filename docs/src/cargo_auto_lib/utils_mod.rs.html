9/// Position of start of the delimited data after the delimiter
10pub fn find_pos_start_data_after_delimiter(md_text_content: &str, pos: usize, delimiter: &str) -> Result<usize> {
11    if let Ok(pos_start_data) = find_from(md_text_content, pos, delimiter) {
12        let pos_start_data = pos_start_data + delimiter.len();
13        return Ok(pos_start_data);
14    }
15    // return
16    Err(Error::ErrorFromStr("not found"))
17}

19/// Position of end of the delimited data before the delimiter
20pub fn find_pos_end_data_before_delimiter(md_text_content: &str, pos: usize, delimiter: &str) -> Result<usize> {
21    if let Ok(pos_end_data) = find_from(md_text_content, pos, delimiter) {
22        return Ok(pos_end_data);
23    }
24    //return
25    Err(Error::ErrorFromStr("not found"))
26}

30/// Find from pos
31pub fn find_from(text: &str, from_pos: usize, find: &str) -> Result<usize> {
32    let slice01 = text.get(from_pos..).ok_or_else(|| Error::ErrorFromStr("text get is None"))?;
33    let option_location = slice01.find(find);
34    if let Some(location) = option_location {
35        // return Ok with usize
36        Ok(from_pos + location)
37    } else {
38        // return Err
39        Err(Error::ErrorFromStr("location is not find"))
40    }
41}

43// region: auto_md_to_doc_comments include doc_comments/traverse_dir_with_exclude_dir.md A ///
44/// Traverse dir and its sub-dir, but avoid excluded dirs.
45///
46/// The find_file and the exclude dir strings must start with /.
47///
48/// ## Example
49///
50/// ```Rust ignore
51///
52/// let files = cargo_auto_lib::traverse_dir_with_exclude_dir(
53///     Path::new("/home/project/src"),
54///     "/*.rs",
55///     // avoid big folders
56///     &vec![
57///         "/.git".to_string(),
58///         "/target".to_string(),
59///         "/docs".to_string()
60///     ]
61/// ).expect("error");
62/// for rs_file_name in files.iter() {
63///     println!("{}", &rs_file_name);
64/// }
65/// ```
66///
67// endregion: auto_md_to_doc_comments include doc_comments/traverse_dir_with_exclude_dir.md A ///
68pub fn traverse_dir_with_exclude_dir(dir: &std::path::Path, find_file: &str, exclude_dirs: &[String]) -> Result<Vec<String>> {
69    // if the parameter is /*.rs, I can eliminate /*
70    let find_file = &find_file.replace("/*", "");
71
72    let mut v = Vec::new();
73    if dir.is_dir() {
74        for entry in std::fs::read_dir(dir)? {
75            let entry = entry?;
76            let path = entry.path();
77            let str_path = path.to_str().ok_or_else(|| Error::ErrorFromStr("path is None"))?;
78            if path.is_dir() {
79                let mut is_excluded = false;
80                for excl in exclude_dirs {
81                    if str_path.ends_with(excl) {
82                        is_excluded = true;
83                        break;
84                    }
85                }
86                if !is_excluded {
87                    let mut sub_v = traverse_dir_with_exclude_dir(&path, find_file, exclude_dirs)?;
88                    v.append(&mut sub_v);
89                }
90            } else if str_path.ends_with(find_file) {
91                v.push(str_path.to_string());
92            }
93        }
94    }
95    Ok(v)
96}

98/// The original `concat()` function does not have a delimiter.
99pub fn concatenate_vec_to_string(vec: &[String], delimiter: &str) -> String {
100    let size = vec.iter().fold(0, |a, b| a + b.len());
101    let mut concatenated_string = String::with_capacity(size);
102    for (i, item) in vec.iter().enumerate() {
103        if i > 0 {
104            concatenated_string.push_str(delimiter);
105        }
106        concatenated_string.push_str(item);
107    }
108    // return
109    concatenated_string
110}
