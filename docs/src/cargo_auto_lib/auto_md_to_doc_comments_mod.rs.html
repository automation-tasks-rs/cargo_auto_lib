35// region: auto_md_to_doc_comments include doc_comments/auto_md_to_doc_comments.md A ///
36/// Finds rs files with markers and include segments from md files as doc comments.  
37///
38/// ![auto_md_to_doc_comments.png](https://github.com/automation-tasks-rs/cargo_auto_lib/blob/main/images/auto_md_to_doc_comments.png?raw=true)
39///
40/// From this doc comments `cargo doc` will generated the documentation and auto-completion.  
41/// We don't want to manually copy this segments. We want them to be automatically in sync.  
42/// We will just run this function before every `cargo doc` with an automation task.  
43/// The `auto_md_to_doc_comments` function must be executed in the project root folder where is the Cargo.toml file.  
44/// First it searches all the rs files in src, tests and examples folders.  
45/// If they contain the markers, than finds the md file and the named segment and include it as doc comments into the rs file.  
46/// The markers are always in pairs: start and end. So exactly the content in between is changed.  
47/// The markers are always comments, so they don't change the code.  
48/// It works only for files with LF line delimiter. No CR and no CRLF.  
49///
50/// ## markers
51///
52/// In the rs file write these markers:  
53///
54/// ```code
55/// //comment region: auto_md_to_doc_comments include README.md A ///
56/// //comment endregion: auto_md_to_doc_comments include README.md A ///
57/// ```
58///
59/// In your rust code, change the word `comment` with double slash `//`.  
60/// In the md file put markers to mark the segment:  
61///
62/// ```markdown
63/// [//comment]: # (auto_md_to_doc_comments segment start A)  
64/// [//comment]: # (auto_md_to_doc_comments segment end A)  
65/// ```
66///
67/// In this instructions I changed `[//]` to `[//comment]` to not process these markers.
68///
69/// The marker must be exclusively in one line. No other text in the same line.  
70/// auto_md_to_doc_comments will delete the old lines between the markers.  
71/// It will find the md file and read the content between the markers.  
72/// Before each line it will add the doc comment symbol as is defined in the marker.  
73/// Finally it will include the new lines as doc comments in the rs file.
74///
75// endregion: auto_md_to_doc_comments include doc_comments/auto_md_to_doc_comments.md A ///
76pub fn auto_md_to_doc_comments() -> Result<()> {
77    let mut cache_md_segments = vec![];
78    for rs_filename in rs_files()?.iter() {
79        let mut rs_text_content = std::fs::read_to_string(rs_filename)?;
80
81        // check if file have CRLF instead of LF and show error
82        if rs_text_content.contains("\r\n") {
83            return Err(Error::ErrorFromString(format!(
84                "{RED}Error: {rs_filename} has CRLF line endings instead of LF. Correct the file! {RESET}"
85            )));
86        }
87
88        let markers = rs_file_markers(&rs_text_content)?;
89        if !markers.is_empty() {
90            for marker in markers.iter().rev() {
91                let segment_text = get_md_segments_using_cache(
92                    &mut cache_md_segments,
93                    &marker.md_filename,
94                    &marker.marker_name,
95                    &marker.comment_symbol,
96                )?;
97                rs_text_content.replace_range(marker.pos_start..marker.pos_end, &segment_text);
98            }
99            println!("  {YELLOW}Write file: {rs_filename}{RESET}");
100            std::fs::write(rs_filename, rs_text_content)?;
101        }
102    }
103    Ok(())
104}

106/// All rs files in src, tests and examples folders.
107fn rs_files() -> Result<Vec<String>> {
108    let mut rs_files = vec![];
109    // in Unix shell ** means recursive match through all the subdirectories
110    for filename_result in glob("src/**/*.rs")? {
111        let filename_pathbuff = filename_result?;
112        let rs_filename = filename_pathbuff
113            .to_str()
114            .ok_or_else(|| Error::ErrorFromStr("filename_pathbuff is None"))?
115            .to_string();
116        rs_files.push(rs_filename);
117    }
118    for filename_result in glob("tests/**/*.rs")? {
119        let filename_pathbuff = filename_result?;
120        let rs_filename = filename_pathbuff
121            .to_str()
122            .ok_or_else(|| Error::ErrorFromStr("filename_pathbuff is None"))?
123            .to_string();
124        rs_files.push(rs_filename);
125    }
126    for filename_result in glob("examples/**/*.rs")? {
127        let filename_pathbuff = filename_result?;
128        let rs_filename = filename_pathbuff
129            .to_str()
130            .ok_or_else(|| Error::ErrorFromStr("filename_pathbuff is None"))?
131            .to_string();
132        rs_files.push(rs_filename);
133    }
134    //return
135    Ok(rs_files)
136}

138lazy_static! {
139    /// Regex for start marker
140    static ref REGEX_RS_START: Regex = Regex::new(r#"(?m)^ *?// region: auto_md_to_doc_comments include (.*?) (.*?) (.*?)$"#).expect("regex new");
141    /// Regex for end marker
142    static ref REGEX_RS_END: Regex = Regex::new(r#"(?m)^ *?// endregion: auto_md_to_doc_comments include (.*?) (.*?) (.*?)$"#).expect("regex new");
143}
144/// markers in rs files
145fn rs_file_markers(rs_text_content: &str) -> Result<Vec<RsMarker>> {
146    let mut markers = vec![];
147    for cap in REGEX_RS_START.captures_iter(rs_text_content) {
148        let rs_marker = RsMarker {
149            md_filename: cap[1].to_string(),
150            marker_name: cap[2].to_string(),
151            comment_symbol: cap[3].to_string(),
152            pos_start: cap.get(0).ok_or_else(|| Error::ErrorFromStr("cap get 0 is None"))?.end() + 1,
153            pos_end: 0,
154        };
155        markers.push(rs_marker);
156    }
157    for cap in REGEX_RS_END.captures_iter(rs_text_content) {
158        // TODO:  error what file and segment are problematic
159        let marker = markers
160            .iter_mut()
161            .find(|m| m.md_filename == cap[1] && m.marker_name == cap[2])
162            .ok_or_else(|| Error::ErrorFromStr("find is None"))?;
163        marker.pos_end = cap.get(0).ok_or_else(|| Error::ErrorFromStr("cap get 0 is None"))?.start();
164    }
165    // return
166    Ok(markers)
167}

169lazy_static! {
170    /// Regex for start marker
171    static ref REGEX_MD_START: Regex = Regex::new(r#"(?m)^\[//\]: # \(auto_md_to_doc_comments segment start (.*?)\)$"#).expect("regex new");
172    /// Regex for end marker
173    static ref REGEX_MD_END: Regex = Regex::new(r#"(?m)^\[//\]: # \(auto_md_to_doc_comments segment end (.*?)\)$"#).expect("regex new");
174}

176/// The first time it is called read the file and extracts all the segments into a cache vector.
177///
178/// Subsequent calls read from the cache.
179fn get_md_segments_using_cache(cache: &mut Vec<MdSegment>, md_filename: &str, marker_name: &str, comment_symbol: &str) -> Result<String> {
180    // check the cache
181    if let Some(_seg) = cache.iter().find(|m| m.md_filename == md_filename) {
182        let segment = cache
183            .iter()
184            .find(|m| m.md_filename == md_filename && m.marker_name == marker_name)
185            .ok_or_else(|| Error::ErrorFromStr("find is None"))?;
186        Ok(segment.text.to_string())
187    } else {
188        // process the file
189        println!("  {YELLOW}Read file: {md_filename}{RESET}");
190        let md_text_content = std::fs::read_to_string(md_filename)?;
191
192        // check if file have CRLF instead of LF and show error
193        if md_text_content.contains("\r\n") {
194            return Err(Error::ErrorFromString(format!(
195                "{RED}Error: {md_filename} has CRLF line endings instead of LF. Correct the file! {RESET}"
196            )));
197        }
198
199        for cap in REGEX_MD_START.captures_iter(&md_text_content) {
200            cache.push(MdSegment {
201                md_filename: md_filename.to_owned(),
202                marker_name: cap[1].to_owned(),
203                pos_start: cap.get(0).ok_or_else(|| Error::ErrorFromStr("cap get 0 is None"))?.end() + 1,
204                pos_end: 0,
205                text: String::new(),
206            });
207        }
208        for cap in REGEX_MD_END.captures_iter(&md_text_content) {
209            // TODO:  error what file and segment are problematic
210            let segment = cache
211                .iter_mut()
212                .find(|m| m.md_filename == md_filename && m.marker_name == cap[1])
213                .ok_or_else(|| Error::ErrorFromStr("find is None"))?;
214            segment.pos_end = cap.get(0).ok_or_else(|| Error::ErrorFromStr("cap get 0 is None"))?.start();
215            // the segment begins with a comment, so don't include the next empty row
216            let mut last_line_was_comment = true;
217            for line in md_text_content[segment.pos_start..segment.pos_end].lines() {
218                if line.starts_with("[//]: # (") {
219                    // don't include md comments
220                    last_line_was_comment = true;
221                } else if last_line_was_comment && line.is_empty() {
222                    // don't include empty line after md comments
223                    last_line_was_comment = false;
224                } else {
225                    last_line_was_comment = false;
226                    segment.text.push_str(comment_symbol);
227                    if !line.is_empty() {
228                        segment.text.push(' ');
229                    }
230                    segment.text.push_str(line);
231                    segment.text.push('\n');
232                }
233            }
234        }
235        // TODO:  error what file and segment are problematic
236        let segment = cache
237            .iter()
238            .find(|m| m.md_filename == md_filename && m.marker_name == marker_name)
239            .ok_or_else(|| Error::ErrorFromStr("find is None"))?;
240        //return
241        Ok(segment.text.to_string())
242    }
