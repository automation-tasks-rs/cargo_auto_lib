39// region: auto_md_to_doc_comments include doc_comments/auto_version_from_date.md A ///
40/// New version from date is written to Cargo.toml and service_worker.js.
41///
42/// In Cargo.toml writes the version as the date `yyyy.mmdd.HHMM` ex. `2019.1221.2359`.  
43/// For non-library projects, the semver specification is not really useful.  
44/// Having the version as the date is just fine for executables and much more human readable.  
45///
46/// ### service_worker.js
47///
48/// Inside the PWA service worker javascript file is also needed to change the version.  
49/// The program searches for `service_worker.js` and modify the version.  
50///
51/// ### no need to change version if no files changed
52///
53/// If src/*.rs or Cargo.toml files are not changed from last compile, than no need to change version.  
54/// The dates of the files will be stored in the file .automation_tasks_rs_file_hashes.json near to Cargo.toml.  
55/// Warning: I don't check if the service worker has changed because it rarely does.  
56/// To know if the projects has changed or not, this function saves the dates of all files into `.automation_tasks_rs_file_hashes.json` near Cargo.toml
57///
58// endregion: auto_md_to_doc_comments include doc_comments/auto_version_from_date.md A ///
59pub fn auto_version_from_date() -> Result<()> {
60    auto_version_from_date_internal(false)
61}
62
63/// Just like auto_version_from_date(), but force the new version even if no files are changed.
64///
65/// It is slower, but easier to understand when deployed.
66pub fn auto_version_from_date_forced() -> Result<()> {
67    auto_version_from_date_internal(true)
68}

74/// Internal function to get version from date
75fn auto_version_from_date_internal(force_version: bool) -> Result<()> {
76    let date = Utc::now();
77    let new_version = version_from_date(date);
78    let vec_of_metadata = read_file_metadata()?;
79    let is_files_equal = if force_version {
80        false
81    } else {
82        let js_struct = read_json_file(".automation_tasks_rs_file_hashes.json")?;
83        are_files_equal(&vec_of_metadata, &js_struct.vec_file_metadata)
84    };
85
86    if !is_files_equal {
87        write_version_to_cargo_and_modify_metadata(&new_version, vec_of_metadata)?;
88    }
89    modify_service_js(&new_version)?;
90    Ok(())
91}

93/// Search for file service_worker.js and modify version
94fn modify_service_js(new_version: &str) -> Result<()> {
95    let start_dir = camino::Utf8Path::new("./");
96    for js_filename in &crate::utils_mod::traverse_dir_with_exclude_dir(
97        start_dir.as_std_path(),
98        "/service_worker.js",
99        &["/.git".to_string(), "/target".to_string()],
100    )? {
101        // println!("{}write version in {}{}", *GREEN, js_filename, *RESET);
102        let mut js_content = std::fs::read_to_string(js_filename)?;
103
104        // check if file have CRLF instead of LF and show error
105        if js_content.contains("\r\n") {
106            return Err(Error::ErrorFromString(format!(
107                "{RED}Error: {js_filename} has CRLF line endings instead of LF. Correct the file! {RESET}"
108            )));
109        }
110
111        let delimiter = r#"const CACHE_NAME = '"#;
112        let delimiter_len = delimiter.len();
113        let option_location = js_content.find(delimiter);
114        if let Some(location) = option_location {
115            let start_version = location + delimiter_len;
116            let option_end_quote = find_from(js_content.as_str(), start_version, r#"';"#);
117            if let Ok(end_version) = option_end_quote {
118                //delete all the characters in between the markers
119                let old_version: String = js_content.drain(start_version..end_version).collect();
120                //println!(r#"old version: "{}""#, old_version.as_str());
121                if new_version != old_version {
122                    println!("  {YELLOW}Modify version: {old_version} -> {new_version}{RESET}");
123                    js_content.insert_str(start_version, new_version);
124                    //println!("{}write file: {}{}", *YELLOW, js_filename, *RESET);
125                    let _x = std::fs::write(js_filename, js_content);
126                }
127            } else {
128                return Err(Error::ErrorFromString(format!("{RED}no end quote for version{RESET}")));
129            }
130        } else {
131            return Err(Error::ErrorFromString(format!("{RED}service_worker.js has no version{RESET}")));
132        }
133    }
134    Ok(())
135}

137/// Write version to Cargo.toml
138fn write_version_to_cargo_and_modify_metadata(new_version: &str, mut vec_of_metadata: Vec<FileMetaData>) -> Result<()> {
139    // println!("{}write version to Cargo.toml{}", *GREEN, *RESET);
140    let cargo_filename = "Cargo.toml";
141    let mut cargo_content = std::fs::read_to_string(cargo_filename)?;
142
143    // check if file have CRLF instead of LF and show error
144    if cargo_content.contains("\r\n") {
145        return Err(Error::ErrorFromString(format!(
146            "{RED}Error: {} has CRLF line endings instead of LF. Correct the file! {RESET}",
147            cargo_filename
148        )));
149    }
150
151    let delimiter = r#"version = ""#;
152    let delimiter_len = delimiter.len();
153    let option_location = cargo_content.find(delimiter);
154    if let Some(location) = option_location {
155        let start_version = location + delimiter_len;
156        let option_end_quote = find_from(cargo_content.as_str(), start_version, r#"""#);
157        if let Ok(end_version) = option_end_quote {
158            //delete all the characters in between the markers
159            let old_version: String = cargo_content.drain(start_version..end_version).collect();
160            //println!(r#"old version: "{}""#, old_version.as_str());
161            if new_version != old_version.as_str() {
162                println!("  {YELLOW}Modify version: {old_version} -> {new_version}{RESET}");
163                cargo_content.insert_str(start_version, new_version);
164                // println!("{}write file: {}{}", *YELLOW, cargo_filename, *RESET);
165                let _x = std::fs::write(cargo_filename, cargo_content);
166
167                //the Cargo.toml is now different
168                correct_file_metadata_for_cargo_tom_inside_vec(&mut vec_of_metadata)?;
169                save_json_file_for_file_meta_data(vec_of_metadata)?;
170            }
171        } else {
172            return Err(Error::ErrorFromString(format!("{RED}no end quote for version{RESET}")));
173        }
174    } else {
175        return Err(Error::ErrorFromString(format!("{RED}Cargo.toml has no version{RESET}")));
176    }
177    Ok(())
178}

180/// Cargo.toml is now different and needs to be changed in the vec of file metadata
181pub fn correct_file_metadata_for_cargo_tom_inside_vec(vec_of_metadata: &mut [FileMetaData]) -> Result<()> {
182    //correct the vector only for Cargo.toml file
183    let filename = "Cargo.toml".to_string();
184    // calculate hash of file
185    let filehash = sha256_digest(std::path::PathBuf::from_str(&filename)?.as_path())?;
186    vec_of_metadata
187        .get_mut(0)
188        .ok_or(Error::ErrorFromStr("error vec_of_metadata.get_mut(0)"))?
189        .filehash = filehash;
190    Ok(())
191}

193/// If files are added or deleted, other files must be also changed
194///
195/// Check if the files on the filesystem are the same as in the json.
196pub fn are_files_equal(vec_of_metadata: &[FileMetaData], js_vec_of_metadata: &[FileMetaData]) -> bool {
197    let mut is_files_equal = true;
198    for x in vec_of_metadata.iter() {
199        //search in json file
200        let mut is_one_equal = false;
201        for y in js_vec_of_metadata.iter() {
202            if x.filename == y.filename && x.filehash == y.filehash {
203                is_one_equal = true;
204                break;
205            }
206        }
207        if !is_one_equal {
208            // println!("{} {}", x.filename, x.filehash);
209            is_files_equal = false;
210            break;
211        }
212    }
213    is_files_equal
214}

216/// Make a vector of file metadata
217pub fn read_file_metadata() -> Result<Vec<FileMetaData>> {
218    let mut vec_of_metadata: Vec<FileMetaData> = Vec::new();
219    let filename = "Cargo.toml".to_string();
220    // calculate hash of file
221    let filehash = sha256_digest(std::path::PathBuf::from_str(&filename)?.as_path())?;
222    vec_of_metadata.push(FileMetaData { filename, filehash });
223
224    let files_paths = crate::utils_mod::traverse_dir_with_exclude_dir(
225        camino::Utf8Path::new("src").as_std_path(),
226        "/*.rs",
227        // avoid big folders
228        &[],
229    )?;
230
231    for filename in files_paths {
232        // calculate hash of file
233        let filehash = sha256_digest(std::path::PathBuf::from_str(&filename)?.as_path())?;
234        vec_of_metadata.push(FileMetaData { filename, filehash });
235    }
236    Ok(vec_of_metadata)
237}

239/// Calculate the hash for the content of a file
240fn sha256_digest(path: &std::path::Path) -> Result<String> {
241    let file = std::fs::File::open(path)?;
242    let mut reader = std::io::BufReader::new(file);
243    // let mut context = ring::digest::Context::new(&ring::digest::SHA256);
244    let mut hasher = sha2::Sha256::new();
245
246    let mut buffer = [0; 1024];
247    use std::io::Read;
248    loop {
249        let count = reader.read(&mut buffer)?;
250        if count == 0 {
251            break;
252        }
253        hasher.update(&buffer[..count]);
254    }
255    let digest = hasher.finalize();
256    let hash_string = data_encoding::HEXLOWER.encode(digest.as_ref());
257    // return
258    Ok(hash_string)
259}

261/// Read .automation_tasks_rs_file_hashes.json
262pub fn read_json_file(json_filepath: &str) -> Result<AutoVersionFromDate> {
263    let js_struct: AutoVersionFromDate;
264    let f = std::fs::read_to_string(json_filepath);
265
266    match f {
267        Ok(x) => {
268            // check if file have CRLF instead of LF. This are unusable - create empty struct
269            if x.contains("\r\n") {
270                //create empty struct
271                js_struct = AutoVersionFromDate {
272                    vec_file_metadata: Vec::new(),
273                }
274            } else {
275                //read struct from file
276                js_struct = serde_json::from_str(x.as_str())?;
277            }
278        }
279        Err(_error) => {
280            // println!("Creating new file: {}", json_filepath);
281            //create empty struct
282            js_struct = AutoVersionFromDate {
283                vec_file_metadata: Vec::new(),
284            }
285        }
286    };
287    Ok(js_struct)
288}

301/// Convert a date to a version
302fn version_from_date(date: DateTime<chrono::Utc>) -> String {
303    // in Rust the version must not begin with zero.
304    // There is an exceptional situation where is midnight 00.
305    //return
306    if date.hour() == 0 {
307        format!("{:04}.{}{:02}.{}", date.year(), date.month(), date.day(), date.minute())
308    } else {
309        format!(
310            "{:04}.{}{:02}.{}{:02}",
311            date.year(),
312            date.month(),
313            date.day(),
314            date.hour(),
315            date.minute()
316        )
317    }
318}

320/// Find from position in string
321fn find_from(rs_content: &str, from: usize, find: &str) -> Result<usize> {
322    let slice01 = rs_content.get(from..).ok_or_else(|| Error::ErrorFromStr("get from is None"))?;
323    let option_location = slice01.find(find);
324    if let Some(location) = option_location {
325        //return Ok with usize
326        Ok(from + location)
327    } else {
328        //return Error
329        Err(Error::ErrorFromStr("location not found"))
330    }
331}
