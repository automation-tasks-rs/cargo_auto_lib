11/// Sync, check, create, push git tag.
12pub fn git_tag_sync_check_create_push(version: &str) -> Result<String> {
13    // sync the local and remote tags
14    crate::run_shell_command_static("git fetch origin --tags --force")?;
15
16    let tags = crate::run_shell_command_output("git tag")?.stdout;
17    let tag_name_version = format!("v{}", &version);
18    if !tags.contains(&format!("{}\n", tag_name_version)) {
19        // create git tag and push
20        crate::ShellCommandLimitedDoubleQuotesSanitizer::new(r#"git tag -f -a "{tag_name_version}" -m "version_{version}" "#)?
21            .arg("{tag_name_version}", &tag_name_version)?
22            .arg("{version}", version)?
23            .run()?;
24
25        crate::run_shell_command_static("git push origin --tags")?;
26    }
27    // return
28    Ok(tag_name_version)
29}

31/// Get release text from RELEASES.md.
32///
33/// First, the user must write the content into file RELEASES.md in the section ## Unreleased.  
34/// Then the automation task will copy the content to GitHub release  
35/// and create a new Version title in RELEASES.md.  
36pub fn body_text_from_releases_md() -> Result<String> {
37    create_releases_md_if_file_not_exist()?;
38    let release_md = std::fs::read_to_string(RELEASES_MD)?;
39    // find the start of ## Unreleased
40    let pos_start_data = crate::find_pos_start_data_after_delimiter(&release_md, 0, "## Unreleased\n")
41        .map_err(|_| Error::ErrorFromStr("## Unreleased is None"))?;
42    // find the beginning of the next ## Version
43    let pos_end_data = crate::find_pos_end_data_before_delimiter(&release_md, pos_start_data, "## Version ")
44        .map_err(|_| Error::ErrorFromStr("## Version is None"))?;
45    let body_md_text = release_md[pos_start_data..pos_end_data - 1].to_string();
46
47    // return
48    Ok(body_md_text)
49}

51/// Create a new Version title in RELEASES.md.
52pub fn create_new_version_in_releases_md(release_name: &str) -> Result<()> {
53    create_releases_md_if_file_not_exist()?;
54    let release_md = std::fs::read_to_string(RELEASES_MD)?;
55    // find the start of ## Unreleased
56    let pos_start_data = crate::find_pos_start_data_after_delimiter(&release_md, 0, "## Unreleased\n")
57        .map_err(|_| Error::ErrorFromStr("## Unreleased is None"))?;
58
59    // create a new Version title after ## Unreleased in RELEASES.md
60    let new_release_md = format!(
61        "{}\n## {}\n{}",
62        &release_md[..pos_start_data],
63        &release_name,
64        &release_md[pos_start_data..]
65    );
66    std::fs::write(RELEASES_MD, new_release_md)?;
67    // return
68    Ok(())
69}

71/// Create RELEASES.md if file not exist
72fn create_releases_md_if_file_not_exist() -> Result<()> {
73    if !camino::Utf8Path::new(RELEASES_MD).exists() {
74        // create the template file
75        let cargo_toml = crate::CargoToml::read()?;
76        let project_name = cargo_toml.package_name();
77        let github_owner = cargo_toml
78            .github_owner()
79            .ok_or_else(|| Error::ErrorFromStr("github_owner is None"))?;
80        let template = format!(
81            r#"# Releases changelog of {project_name}
82
83All notable changes to this project will be documented in this file.  
84This project adheres to [Semantic Versioning](https://semver.org/spec/v2.0.0.html).  
85The library releases will be published on crates.io.  
86The cargo-auto automation task will use the content of the section `## Unreleased` to create
87the GitHub release consistently with this file.  
88The ongoing changes that are not released, are visible in the git commits and GitHub pull requests.  
89The TODO section is part of the [README.md](https://github.com/{github_owner}/{project_name}).  
90
91## Unreleased
92
93## Version 0.0.1
94
95"#
96        );
97        std::fs::write(RELEASES_MD, template)?;
98    }
99    Ok(())
100}

102/// Add commit message to Unreleased in RELEASES.md.
103pub fn add_message_to_unreleased(message: &str) -> Result<()> {
104    create_releases_md_if_file_not_exist()?;
105    let release_md = std::fs::read_to_string(RELEASES_MD)?;
106    // find the beginning of the first ## Version
107    let Ok(pos_end_data) = crate::find_pos_end_data_before_delimiter(&release_md, 0, "## Version ") else {
108        return Ok(());
109    };
110    // add before the first ## Version
111    // I expect only one empty line before ## Version
112    let added_message_md = format!("{}- {}\n\n{}", &release_md[..pos_end_data], message, &release_md[pos_end_data..]);
113    std::fs::write(RELEASES_MD, added_message_md)?;
114    Ok(())
