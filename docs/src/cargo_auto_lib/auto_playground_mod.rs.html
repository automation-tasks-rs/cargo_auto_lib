3//! Includes the link to playground with the rust code in a parameter.
4
5use crate::error_mod::{Error, Result};
6use crate::public_api_mod::{RED, RESET, YELLOW};
7use crate::utils_mod::*;
8use lazy_static::lazy_static;
9
10lazy_static! {
11    /// Capture the link in markdown style [name](link)
12    static ref REGEX_MD_LINK: regex::Regex = regex::Regex::new(
13    r#".+\[.+\]\((.+)\).+"#
14    ).expect("regex new");
15}
16
17// region: auto_md_to_doc_comments include doc_comments/auto_playground_run_code.md A ///
18/// Include the link to run code in Rust playground.
19///
20/// The function searches in all markdown files for markers like this:
21///
22/// ```markdown
23/// [//comment]: # (auto_playground start)
24///
25/// Run this code in the [Rust playground](https://play.rust-lang.org/?version=stable&mode=debug&edition=2021&code=fn%20m%0A%7D):
26///
27/// '''Rust ignore
28/// fn main(){
29///     println!("Hello World!");
30/// }
31/// '''
32///
33/// [//comment]: # (auto_playground end)
34/// ```
35///
36/// In this instructions I changed `[//]` to `[//comment]` and  ticks to single quotes to not process these markers.
37///
38/// Between the start marker and the first triple backtick there is the link in "()" parentheses. The link to Rust playground encodes the code with url_encoding (percents) and sends it as an Url parameter.
39///
40/// Info: docs.rs has already a functionality that shows the `Run` button on your code and can run code the playground if you put this line at the top of lib.rs:
41///
42/// ``` Rust ignore
43/// #![doc(html_playground_url = "https://play.rust-lang.org")]
44/// ```
45///
46/// But it works only on docs.rs.  
47/// I want to run my code examples from everywhere: from GitHub README.md, GitHub pages and crates.io.  
48///
49// endregion: auto_md_to_doc_comments include doc_comments/auto_playground_run_code.md A ///
50pub fn auto_playground_run_code() -> Result<()> {
51    println!("  {YELLOW}Running auto_playground{RESET}");
52    let path = std::env::current_dir()?;
53    //use traverse instead of glob
54    let files = crate::utils_mod::traverse_dir_with_exclude_dir(
55        &path,
56        "/*.md",
57        // exclude folders
58        &["/.git".to_string(), "/target".to_string(), "/docs".to_string(), "/pkg".to_string()],
59    )?;
60    for md_filename in files {
61        let md_filename = camino::Utf8Path::new(&md_filename);
62
63        let mut is_changed = false;
64        let md_old = std::fs::read_to_string(md_filename)?;
65
66        // check if file have CRLF and show error
67        if md_old.contains("\r\n") {
68            return Err(Error::ErrorFromString(format!(
69                "{RED}Error: {md_filename} has CRLF line endings instead of LF. Correct the file! {RESET}"
70            )));
71        }
72        let mut iteration_start_pos = 0;
73        let mut md_new = String::new();
74        // find markers
75        while let Ok(marker_start) =
76            find_pos_start_data_after_delimiter(&md_old, iteration_start_pos, "\n[//]: # (auto_playground start)\n")
77        {
78            let Ok(code_start) = find_pos_start_data_after_delimiter(&md_old, marker_start, "\n```") else {
79                return Ok(());
80            };
81            // the first triple tick can be ``` or ```rust or ```Rust or ```Rust ignore. Therefore I find the triple tick and then I find next newline.
82            let Ok(code_start) = find_pos_start_data_after_delimiter(&md_old, code_start, "\n") else {
83                return Ok(());
84            };
85            let Ok(code_end) = find_pos_end_data_before_delimiter(&md_old, code_start + 3, "\n```\n") else {
86                return Ok(());
87            };
88            let Ok(marker_end) = find_pos_end_data_before_delimiter(&md_old, marker_start, "\n[//]: # (auto_playground end)\n") else {
89                return Ok(());
90            };
91            let rust_code = &md_old[code_start..code_end];
92            let rust_code_encoded = urlencoding::encode(rust_code).to_string();
93            let playground_link = format!("https://play.rust-lang.org/?version=stable&mode=debug&edition=2021&code={rust_code_encoded}");
94            // Some browsers can limit the url length to 2048 characters. That is absolutely too short for any interesting code example.
95            // TODO: in that case save a gist with GitHub API and send that gist_id to playground? That can be very long.
96
97            // replace the link inside markdown link notation. First find the text between marker_start and code_start
98            let text_that_has_the_link = &md_old[marker_start..code_start];
99            // parse this format [Rust playground](https:...)
100            let cap_group = REGEX_MD_LINK.captures(text_that_has_the_link).ok_or_else(|| {
101                Error::ErrorFromString(format!(
102                    "{RED}Error: The old link '{text_that_has_the_link}' is NOT in this format '[Rust playground](https:...)'{RESET}"
103                ))
104            })?;
105            let old_link = &cap_group[1];
106            // replace the old link with the new one
107            let text_that_has_the_link = text_that_has_the_link.replace(old_link, &playground_link);
108            is_changed = true;
109            md_new.push_str(&format!(
110                "{}{}{}",
111                &md_old[iteration_start_pos..marker_start],
112                &text_that_has_the_link,
113                &md_old[code_start..marker_end],
114            ));
115
116            // start of the next iteration
117            iteration_start_pos = marker_end;
118        } // end while
119
120        // if changed, then write to disk
121        if is_changed {
122            println!("  {YELLOW}Code inside {md_filename} has changed. Playground link corrected.{RESET}");
123            // push the remaining text
124            md_new.push_str(&md_old[iteration_start_pos..md_old.len()]);
125            let bak_filename = md_filename.with_extension("bak");
126            std::fs::write(&bak_filename, md_old)?;
127            std::fs::write(md_filename, md_new)?;
128        }
129    }
130    println!("  {YELLOW}Finished auto_playground{RESET}");
131    Ok(())
132}
