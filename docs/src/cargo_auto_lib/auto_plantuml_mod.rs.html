15        r#"(?m)^\[//\]: # \(auto_plantuml start\)$"#

27// region: auto_md_to_doc_comments include doc_comments/auto_plantuml.md A ///
28/// Includes the plantuml svg generated from the plantuml code.
29///
30/// Search for markers in md files and process plantuml code.  
31///
32/// ```markdown
33/// [//comment]: # (auto_plantuml start)
34///
35/// '''plantuml
36///     @startuml
37///     [Bob] ..> [Alice]
38///     @enduml
39/// '''
40///
41/// ![svg_534231](images/svg_534231.svg)  
42///
43/// [//comment]: # (auto_plantuml end)
44/// ```
45///
46/// In this instructions I changed `[//]` to `[//comment]` and  ticks to single quotes to not process these markers.
47///
48/// Between the last triple backtick and the end marker is the processed svg file.  
49/// Calculate a short hash from the plantuml code.  
50/// If the name of the svg file contains this hash code it means, we already have the correct svg file.  
51/// Else we have to delete the old svg file and get a new one from the modified plantuml code.  
52/// Call the plantuml.com server with the plantuml code and saves the result svg file in folder images/.  
53/// Add the hash code to the filename.
54///
55/// Process plantuml in current directory.  
56/// Finds markers (auto_plantuml start) and (auto_plantuml end) in md files.  
57/// If needed calls the web service and saves the svg file.  
58/// Between markers adds the link to the svg file.  
59/// repo_url like <https://github.com/automation-tasks-rs/sey_currency_converter_pwa>
60/// So the image file link is from the repository and accessible everywhere.
61///
62// endregion: auto_md_to_doc_comments include doc_comments/auto_plantuml.md A ///
63pub fn auto_plantuml(repo_url: &str) -> Result<()> {
64    let path = std::env::current_dir()?;
65    auto_plantuml_for_path(&path, repo_url)?;
66    Ok(())
67}

69/// Process plantuml for all md files.
70///
71/// For test and examples I need to provide the path.
72pub fn auto_plantuml_for_path(path: &std::path::Path, repo_url: &str) -> Result<()> {
73    let path = camino::Utf8Path::from_path(path).ok_or_else(|| Error::ErrorFromStr("from_path is None"))?;
74    println!("  {YELLOW}Running auto_plantuml{RESET}");
75    //use traverse instead of glob
76    let files = crate::utils_mod::traverse_dir_with_exclude_dir(
77        path.as_std_path(),
78        "/*.md",
79        // avoid big folders
80        &["/.git".to_string(), "/target".to_string(), "/docs".to_string()],
81    )?;
82    for md_filename in files {
83        let md_filename = camino::Utf8Path::new(&md_filename);
84
85        let mut is_changed = false;
86        let mut md_text_content = std::fs::read_to_string(md_filename)?;
87
88        // check if file have CRLF and show error
89        if md_text_content.contains("\r\n") {
90            return Err(Error::ErrorFromString(format!(
91                "{RED}Error: {md_filename} has CRLF line endings instead of LF. Correct the file! {RESET}"
92            )));
93        }
94        let mut pos = 0;
95        // find markers
96        while let Ok(marker_start) = find_pos_start_data_after_delimiter(&md_text_content, pos, "\n[//]: # (auto_plantuml start)\n") {
97            pos = marker_start + 34;
98            if let Ok(code_start) = find_pos_start_data_after_delimiter(&md_text_content, marker_start, "\n```plantuml\n") {
99                if let Ok(code_end) = find_pos_end_data_before_delimiter(&md_text_content, code_start, "\n```\n") {
100                    let code_end_after = code_end + 5;
101                    let plantuml_code = &md_text_content[code_start..code_end];
102                    let plantuml_code_hash = hash_text(plantuml_code);
103                    if let Ok(marker_end) =
104                        find_pos_end_data_before_delimiter(&md_text_content, marker_start, "\n[//]: # (auto_plantuml end)\n")
105                    {
106                        let img_link = md_text_content[code_end_after..marker_end].trim();
107                        let mut get_new_svg = false;
108                        if img_link.is_empty() {
109                            get_new_svg = true;
110                        } else {
111                            // parse this format ![svg_534231](images/svg_534231.svg)
112                            let cap_group = REGEX_IMG_LINK
113                                .captures(img_link)
114                                .ok_or_else(|| Error::ErrorFromString(format!("{RED}Error: The old img link '{img_link}' is NOT in this format '![svg_534231](images/svg_534231.svg)'{RESET}")))?;
115                            let old_hash = &cap_group[1];
116                            if old_hash != plantuml_code_hash {
117                                get_new_svg = true;
118                                // delete the old image file
119                                let old_file_path = camino::Utf8PathBuf::from_str(&format!(
120                                    "{}/images/svg_{old_hash}.svg",
121                                    md_filename.parent().ok_or_else(|| Error::ErrorFromStr("parent is None"))?
122                                ))?;
123                                if old_file_path.exists() {
124                                    std::fs::remove_file(&old_file_path)?;
125                                }
126                            } else {
127                                // check if the svg file exists
128                                let old_file_path = camino::Utf8PathBuf::from_str(&format!(
129                                    "{}/images/svg_{old_hash}.svg",
130                                    md_filename.parent().ok_or_else(|| Error::ErrorFromStr("parent is None"))?
131                                ))?;
132                                if !old_file_path.exists() {
133                                    get_new_svg = true;
134                                }
135                            }
136                        }
137                        if get_new_svg {
138                            let relative_md_filename = md_filename.strip_prefix(path)?;
139                            println!("  {YELLOW}{relative_md_filename} get new svg {plantuml_code_hash}{RESET}");
140
141                            // get the new svg image
142                            let svg_code = request_svg(plantuml_code)?;
143                            let new_file_path = camino::Utf8PathBuf::from_str(&format!(
144                                "{}/images/svg_{plantuml_code_hash}.svg",
145                                md_filename.parent().ok_or_else(|| Error::ErrorFromStr("parent is None"))?
146                            ))?;
147                            std::fs::create_dir_all(new_file_path.parent().ok_or_else(|| Error::ErrorFromStr("parent is None"))?)?;
148                            std::fs::write(&new_file_path, svg_code)?;
149                            // if repo_url is not empty then prepare GitHub url
150                            // https://github.com/automation-tasks-rs/sey_currency_converter_pwa/raw/main/
151                            let repo_full_url = if repo_url.is_empty() {
152                                "".to_string()
153                            } else {
154                                format!("{}/raw/main/", repo_url.trim_end_matches('/'))
155                            };
156                            // path relative to Cargo.toml (project root)
157                            let relative_svg_path = new_file_path.strip_prefix(path)?;
158                            // create the new image lnk
159                            let img_link = format!("\n![svg_{plantuml_code_hash}]({repo_full_url}{relative_svg_path})\n");
160                            // delete the old img_link and insert the new one
161                            md_text_content.replace_range(code_end_after..marker_end, &img_link);
162                            is_changed = true;
163                        }
164                    }
165                }
166            }
167        }
168        // if changed, then write to disk
169        if is_changed {
170            std::fs::write(md_filename, md_text_content)?;
171        }
172    }
173    println!("  {YELLOW}Finished auto_plantuml{RESET}");
174    Ok(())
175}

177/// Hash text.
178pub fn hash_text(text: &str) -> String {
179    let hash = <sha2::Sha256 as sha2::Digest>::digest(text.as_bytes());
180    // base64ct = {version = "1.5.0", features = ["alloc"] }
181    // return base64_hash
182    <base64ct::Base64UrlUnpadded as base64ct::Encoding>::encode_string(&hash)
183}

185/// Request svg from plantuml server
186pub fn request_svg(plant_uml_code: &str) -> Result<String> {
187    let base_url = "https://plantuml.com/plantuml";
188    let url_parameter = compress_plant_uml_code(plant_uml_code)?;
189    let url = format!("{}/svg/{}", base_url, url_parameter);
190    // use reqwest to GET from plantuml.com server
191    // return response
192    Ok(reqwest::blocking::get(url)?.text_with_charset("utf-8")?)
193}

195/// Deflate and strange base64, that is Url_safe
196pub fn compress_plant_uml_code(plant_uml_code: &str) -> Result<String> {
197    // first deflate
198    let data = plant_uml_code.as_bytes();
199    let compressed = deflate::deflate_bytes(data);
200    // then the strange base64
201    // https://plantuml.com/text-encoding
202    let my_cfg = radix64::CustomConfig::with_alphabet("0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz-_")
203        .no_padding()
204        .build()
205        .map_err(|_| Error::ErrorFromStr("CustomConfigError"))?;
206    // return
207    Ok(my_cfg.encode(&compressed))
208}
