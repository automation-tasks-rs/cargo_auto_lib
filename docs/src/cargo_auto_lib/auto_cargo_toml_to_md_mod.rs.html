20lazy_static! {
21    /// Regex for start marker
22    static ref REGEX_MD_START: Regex = Regex::new(r#"(?m)^\[//\]: # \(auto_cargo_toml_to_md start\)$"#).expect("regex new");
23    /// Regex for end marker
24    static ref REGEX_MD_END: Regex = Regex::new(r#"(?m)^\[//\]: # \(auto_cargo_toml_to_md end\)$"#).expect("regex new");
25}

27// region: auto_md_to_doc_comments include doc_comments/auto_cargo_toml_to_md.md A ///
28/// This function includes data from Cargo.toml to markdown files.  
29///
30/// ![auto_cargo_toml_to_md.png](https://github.com/automation-tasks-rs/cargo_auto_lib/blob/main/images/auto_cargo_toml_to_md.png?raw=true)
31///
32/// This is nice for avoiding out of sync data.  
33/// Run it on every build with `automation_tasks_rs` and [cargo auto](https://crates.io/crates/cargo-auto).  
34///   
35/// In the md file write these markers in invisible markdown comments.
36///
37/// ```markdown
38/// [//comment]: # (auto_cargo_toml_to_md start)
39///
40/// [//comment]: # (auto_cargo_toml_to_md end)
41///
42/// ```
43///
44/// In this instructions I changed `[//]` to `[//comment]` to not process these markers.
45///
46/// `auto_cargo_toml_to_md` deletes the old lines between the markers and includes the Cargo.toml data:  
47/// description, repository, version, utc_now, authors and creates badges for keywords and categories.
48///
49/// The words topics, keywords, hashtags and tags all mean the same concept.  
50/// In cargo.toml we have keywords.  
51/// In README.md I want to have badges with different color. And hashtags for SEO.  
52/// In GitHub they are topics.
53///
54/// Some keywords have defined colors, others are orange like Rust.  
55/// This can be expanded in the future.  
56///
57/// - Yellow: work-in-progress
58/// - Green: maintained, ready-for-use
59/// - Red: obsolete, archived
60///
61// endregion: auto_md_to_doc_comments include doc_comments/auto_cargo_toml_to_md.md A ///
62pub fn auto_cargo_toml_to_md() -> Result<()> {
63    let cargo_toml = crate::auto_cargo_toml_mod::CargoToml::read()?;
64    let version = cargo_toml.package_version();
65    let author_name = cargo_toml.package_author_name();
66    let homepage = cargo_toml.package_homepage();
67    let repository = cargo_toml.package_repository().unwrap_or("".to_owned());
68    let description = cargo_toml.package_description().unwrap_or("".to_owned());
69    let keywords = cargo_toml.package_keywords().to_vec();
70    let now_utc_date_iso = &crate::utils_mod::now_utc_date_iso();
71
72    let mut new_text = format!("\n**{description}**  \n");
73    new_text.push_str(&format!(
74        "***version: {version} date: {now_utc_date_iso} author: [{author_name}]({homepage}) repository: [GitHub]({repository})***\n\n"
75    ));
76
77    for keyword in keywords.iter() {
78        let color = if keyword == "work-in-progress" {
79            "yellow"
80        } else if keyword == "maintained" || keyword == "ready-for-use" {
81            "green"
82        } else if keyword == "obsolete" || keyword == "archived" {
83            "red"
84        } else {
85            "orange"
86        };
87        // inside the shield badge syntax, hyphens must be replaced by underscore
88        let keyword_underscore = keyword.replace('-', "_");
89        new_text.push_str(&format!(
90            " ![{keyword}](https://img.shields.io/badge/{keyword_underscore}-{color})\n"
91        ));
92    }
93    new_text.push('\n');
94
95    for filename_result in glob("*.md")? {
96        let filename_pathbuff = filename_result?;
97        let md_filename = filename_pathbuff
98            .to_str()
99            .ok_or_else(|| Error::ErrorFromStr("filename_pathbuff is None"))?;
100        // println!("checking md_filename: {}", &md_filename);
101        let mut md_text_content = std::fs::read_to_string(md_filename)?;
102
103        // check if file have CRLF and show error
104        if md_text_content.contains("\r\n") {
105            return Err(Error::ErrorFromString(format!(
106                "{RED}Error: {md_filename} has CRLF line endings instead of LF. Correct the file! {RESET}"
107            )));
108        }
109
110        if let Some(cap) = REGEX_MD_START.captures(&md_text_content) {
111            let pos_start = cap.get(0).ok_or_else(|| Error::ErrorFromStr("cap get 0 is None"))?.end() + 1;
112            if let Some(cap) = REGEX_MD_END.captures(&md_text_content) {
113                let pos_end = cap.get(0).ok_or_else(|| Error::ErrorFromStr("cap get 0 is None"))?.start();
114                md_text_content.replace_range(pos_start..pos_end, &new_text);
115                println!("  {YELLOW}Write to md file: {}{RESET}", md_filename);
116                println!("{GREEN}{}{RESET}", &new_text.trim_end_matches("\n\n"));
117                std::fs::write(md_filename, md_text_content)?;
118            }
119        }
120    }
121    Ok(())
122}
